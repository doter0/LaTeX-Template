\usepackage[tmargin=2cm,rmargin=1in,lmargin=1in,bmargin=2cm]{geometry}
\usepackage{fancyhdr}       % page header and footer
\usepackage{amsthm}         % theorem environments
\usepackage{amsmath}        % math environments
\usepackage{amssymb}        % \square 사용.
\usepackage{mdframed}       % boxed frames
\usepackage{xifthen}        % ifthenelse 조건문 사용
\usepackage{enumitem}       % enumerate 설정

% enumerate setting
\setlist[enumerate,1]{label=(\arabic*)}

% fancyhdr 설정
\pagestyle{fancy}
\fancyhf{}
\fancyfoot[R]{\thepage}
\fancyfoot[C]{\textbf{\leftmark}}

% Remove the existing proof environment from amsthm
\let\proof\relax
\let\endproof\relax

 % amsthm의 기본 italic체가 아닌 기본 폰트를 사용하게 함.
\theoremstyle{definition}

\newmdtheoremenv[
    innertopmargin=0em
]{corollary}{Corollary}[chapter]

\newmdtheoremenv[
    innertopmargin=0em
]{lemma}{Lemma}[chapter]

\newmdtheoremenv[
    innertopmargin=0em
]{proposition}{Proposition}[chapter]

\newtheorem*{remark}{Remark}

% Define theorem enviromenet
\newmdtheoremenv[
    innertopmargin=0em,
    skipabove=\baselineskip
    ]{customtheorem}{Theorem}[chapter]

\newenvironment{theorem}[1]{
    \begin{customtheorem}
    \textbf{\ifthenelse{\isempty{#1}}{}{(#1)}}%
}{
    \end{customtheorem}
}

% Define definition environment
\newmdtheoremenv[
    innertopmargin=0em,
    skipabove=\baselineskip
    ]{customdefinition}{Definition}[chapter]

\newenvironment{definition}[1]{
    \begin{customdefinition}
    \textbf{\ifthenelse{\isempty{#1}}{}{(#1)}}%
}{
    \end{customdefinition}
}

% Define example enviromenet
\newenvironment{example}[1][]
{
    \par\noindent\textbf{Example.} \ignorespaces
}{
    \hfill$\Diamond$
    \par
}

% Property 환경 정의
\newenvironment{property}
{
    \vspace{1em}
    \noindent\hrule\vspace{0.5em}\textbf{Property.}\par\vspace{0.5em}\itshape
}
{
    \vspace{0.5em}\hrule\vspace{1em}\normalfont
}

% Identity 환경 정의
\newenvironment{identity}
{
    \vspace{1em}
    \noindent\hrule\vspace{0.5em}\textbf{Identity.}\par\vspace{0.5em}\itshape
}
{
    \vspace{0.5em}\hrule\vspace{1em}\normalfont
}

% Define proof environment
\newmdenv[
    leftline=true,
    topline=false,
    bottomline=false,
    rightline=false,
    linecolor=black,
    innertopmargin=0.5em,
    innerbottommargin=0.5em,
    skipabove=\baselineskip,
    skipbelow=\baselineskip
]{proofbox}

\newenvironment{proof}{
    \begin{proofbox}
    \textbf{Proof.}
}{
    \hfill $\square$
    \end{proofbox}
}
